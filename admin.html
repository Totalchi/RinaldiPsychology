Supabase not configured To connect real appointment requests:
  1. Create a Supabase project at supabase.com
  2. Run the SQL below to create the required tables
  3. Add your supabaseUrl and supabaseAnonKey to ADMIN_CONFIG in this file and ER_CONFIG in js/main.js
-- Appointment requests CREATE TABLE appointment_requests ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), nome TEXT, cognome TEXT, email TEXT, tel TEXT, motivo TEXT, messaggio TEXT, lang TEXT DEFAULT 'it', status TEXT DEFAULT 'new', admin_notes TEXT ); ALTER TABLE appointment_requests ENABLE ROW LEVEL SECURITY; CREATE POLICY "Anon insert" ON appointment_requests FOR INSERT WITH CHECK (true); CREATE POLICY "Auth read" ON appointment_requests FOR SELECT USING (auth.role()='authenticated'); CREATE POLICY "Auth update" ON appointment_requests FOR UPDATE USING (auth.role()='authenticated'); -- Reviews CREATE TABLE reviews ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), author TEXT NOT NULL, role_text TEXT, quote TEXT NOT NULL, stars INT DEFAULT 5, is_visible BOOLEAN DEFAULT true, sort_order INT DEFAULT 0 ); ALTER TABLE reviews ENABLE ROW LEVEL SECURITY; CREATE POLICY "Anon read" ON reviews FOR SELECT USING (is_visible=true); CREATE POLICY "Auth full" ON reviews USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated'); -- Analytics CREATE TABLE analytics_events ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), page TEXT, event_type TEXT DEFAULT 'pageview', session_id TEXT, lat FLOAT, lng FLOAT, city TEXT, region TEXT, country TEXT, browser TEXT, os TEXT, referrer TEXT ); ALTER TABLE analytics_events ENABLE ROW LEVEL SECURITY; CREATE POLICY "Anon insert" ON analytics_events FOR INSERT WITH CHECK (true); CREATE POLICY "Auth read" ON analytics_events FOR SELECT USING (auth.role()='authenticated');
